\begin{exercise}

Evaluate the following expressions.
\begin{align*}
\int_{-\frac{\pi}{2}}^\frac{\pi}{2} \sin^5 x \d x &= \answer{0}\\
\dd{x}\left[\int_0^\frac{\pi}{2} \sin^5 t \d t\right] &= \answer{0}\\
\dd{x}\left[\int_0^{x^3} \sin^5 t \d t\right] &= \answer{3x^2\left(\sin\left(x^3\right)\right)^5}\\
\end{align*}

Evaluate
\[
\dd{x}\left[\int_0^{\cos x} \tan\left(t^2-1\right) \d t\right].
\]

\begin{multipleChoice}
\choice{$\tan\left(x^2-1\right)$}
\choice{$\tan\left(\cos^2(x)-1\right)$}
\choice{$2x\sec^2\left(x^2-1\right)$}
\choice{$\tan\left(\cos^2(x)-1\right) + C$}
\choice[correct]{$-\sin(x)\tan\left(\cos^2(x)-1\right)$}
\choice{$2\cos(x)\sec^2\left(\cos^2(x)-1\right)$}
\choice{None of the above answers}
\end{multipleChoice}

\end{exercise}
