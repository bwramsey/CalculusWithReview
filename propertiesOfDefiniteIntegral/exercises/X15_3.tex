\begin{exercise}
Suppose the function $f$ is the solution to the initial value problem:
\[  \begin{cases}
	y' = x^3 - 4x^2\\
	y(0) = 3
\end{cases}\]
At which $x$-coordinate does $f$ have a local minimum?

\[ \text{ At } x = \answer{4} \]
\begin{hint}
	Find critical points and use the first/second derivative test.
\end{hint}
\end{exercise}
