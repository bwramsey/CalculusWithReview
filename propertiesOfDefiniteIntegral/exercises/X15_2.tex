\begin{exercise}
Rewrite as algebraic expression, for applicable $x>0$.
\begin{align*}
	\sin\left( \cos^{-1}\left(\frac{x}{4}\right) \right) &= \answer{\frac{\sqrt{16-x^2}}{4}} \\ \\
	\cot\left( \csc^{-1}\left( \frac{3x}{5} \right) \right) &= \answer{\frac{5}{\sqrt{9x^2-25}}}
\end{align*}	


\end{exercise}
