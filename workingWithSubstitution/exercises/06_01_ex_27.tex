\begin{exercise}

Find the indefinite integral of $\tan^{2}(x)$ with respect to $x$.

\[
\int \tan^2(x)\d x=\answer{\tan(x)-x+C}
\]
\begin{hint}
	Think about a trig identity. 
	\[ \tan^2(x) + 1 = \answer{\sec^2(x)} \]
\end{hint}

\end{exercise}
