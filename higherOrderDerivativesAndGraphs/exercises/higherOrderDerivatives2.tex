\begin{exercise}

	Let $g(t) = \dfrac{x}{x+3}$.

	Find the derivative.  \[ \ddt g(t) = \answer{\frac{3}{(x+3)^2}} \]
	\begin{exercise}
		Find the second derivative.	
		\[ g''(t) = \answer{\frac{-6}{(x+3)^3}} \]
	\end{exercise}

\end{exercise}
