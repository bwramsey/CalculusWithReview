\outcome{Recognize a composition of functions.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\begin{document}
\begin{exercise}
Let $y(x) = \cos \left(2 \sqrt[6]{3} \sqrt[6]{x}\right)$. Compute:

\[
\dd{x}y(x)
=\answer{\frac{\sin \left(2 (3^{1/6}) x^{1/6}\right)}{3^{5/6} x^{5/6} }}
\]
\end{exercise}
