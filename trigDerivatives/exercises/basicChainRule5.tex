\outcome{Recognize a composition of functions.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\begin{document}
\begin{exercise}
Let $y(x) = \cos \left(2 \sqrt{3} \sqrt{x}\right)$. Compute:

\[
\dd{x}y(x)
=\answer{\frac{-(2^{1/2})\sin \left(2 (3^{1/2}) x^{1/2}\right)}{x^{1/2} }}
\]
\end{exercise}
