\outcome{Recognize a composition of functions.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\begin{document}
\begin{exercise}
	Let $c(t) = -e^{3} \ln\left(\cot (2 t )\right)$. Compute:
	\[ \ddt c(t) =\answer{2e^{3} \frac{\csc^2( 2 t)}{\cot (2 t)}} \]
\end{exercise}
