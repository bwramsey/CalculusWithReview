\begin{exercise}
	Let $c(\theta) = -e^{3} \ln\left(\cot (2 \theta )\right)$. Compute:
	\[ \dd{\theta}c(\theta) =\answer{2e^{3} \frac{\csc^2( 2 \theta)}{\cot (2 \theta )}} \]
\end{exercise}
