\begin{exercise}
	Let $r(v) = \frac{e^v}{4+\cos(e^v)}$. Compute:
	\[ r'(v) =\answer{ \frac{e^v(4+\cos(e^v) + \sin(e^v))}{(4+\cos(e^v))^2}  } \]
\end{exercise}
