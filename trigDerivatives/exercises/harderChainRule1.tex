\begin{exercise}
Let $s(u) = \cos \left(3 \cos \left(2 \sqrt{3} \sqrt{u}\right)\right)$. Compute:

\[
\dd{u}s(u)
\begin{prompt}=\answer{\frac{3 \sqrt{3} \sin \left(2 \sqrt{3} \sqrt{u}\right) \sin \left(3 \cos \left(2 \sqrt{3} \sqrt{u}\right)\right)}{\sqrt{u}}}\end{prompt}
\]
\end{exercise}
