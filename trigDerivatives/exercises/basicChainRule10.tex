\begin{exercise}
Let $p(v) = 3^{3/4} \sqrt[4]{-v}$. Compute:

\[
\dd{v}p(v)
\begin{prompt}=\answer{-\frac{3^{3/4}}{4 (-v)^{3/4}}}\end{prompt}
\]
\end{exercise}
