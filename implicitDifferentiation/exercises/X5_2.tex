		\begin{multipleChoice}

			\choice[correct]{explicit function of $x$.}
		\end{multipleChoice}

		\begin{exercise}
			Use implicit differentiation to find $\frac{dy}{dx}$ where
				\[ x^2+3xy=1 \]
			\[ \frac{dy}{dx} = \answer{-\frac{2x+3y}{3x}} \]
		\end{exercise}
	\end{exercise}
