\begin{exercise}

Consider
\[
3x^2+4xy-6x+y^2-2y=0.
\]
Compute
\[
\frac{dy}{dx} \begin{prompt}=\answer{\frac{3-3x-2y}{2x+y-1}}.
\end{prompt}.
\]
Select all that apply:
\begin{selectAll}
	\choice[correct]{$\frac{d^2y}{dx^2} = 0$}
	\choice[correct]{ $ \frac{d^2y}{dx^2} =\frac{3x^2+4xy-6x+y^2-2y}{(2x+y-1)^3}$}
	\choice{$\frac{d^2y}{dx^2} = \frac{1}{(2x+y-1)^2}$}
	\choice[correct]{ $ \frac{d^2y}{dx^2} =\frac{3x^2+4xy-6x+y^2-2y}{(2x+y-1)^2}$}
\end{selectAll}

\end{exercise}
