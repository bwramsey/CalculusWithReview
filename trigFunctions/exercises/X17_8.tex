\begin{exercise}

	Solve the equation:
	 \[\tan^2(x) = 1-\sec(x) \]
	
	There are $\answer{3}$ solutions in the interval $[0, 2\pi)$.	
	\begin{exercise}
		Write the solutions, in order from smallest to largest.
		
		\begin{align*}
			x & = \answer{0}, \\
			 	& \answer{\frac{2\pi}{3}}, \\
			 	& \answer{\frac{4\pi}{3}}
		\end{align*}	 	 


	\end{exercise}

\end{exercise}
