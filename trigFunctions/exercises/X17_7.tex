\begin{exercise}

	Solve the equation:
	 \[\sin(2x) = \cos(x) \]
	
	There are $\answer{4}$ solutions in the interval $[0, 2\pi)$.	
	\begin{exercise}
		Write the solutions, in order from smallest to largest.
		
		\begin{align*}
			x & = \answer{\frac{\pi}{6}}, \\
			 	& \answer{\frac{\pi}{2}}, \\
			 	& \answer{\frac{5\pi}{6}}, \\
			 	 &\answer{\frac{3\pi}{2}}
		\end{align*}	 	 


	\end{exercise}

\end{exercise}
