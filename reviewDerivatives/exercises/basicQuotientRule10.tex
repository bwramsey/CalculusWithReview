\begin{exercise}
\outcome{Use the quotient rule to calculate derivatives.}
\tag{derivative}
Let $f(y) = \frac{ 3 \sqrt{y }-5 e^{y }}{-3 y ^4}$. Compute:
\[
f'(y)
\begin{prompt}=\answer{\frac{4 \left(3 \sqrt{y }-5 e^{y }\right)}{3 y ^5}-\frac{\frac{3}{2 \sqrt{y }}-5 e^{y }}{3 y ^4}}\end{prompt}
\]
\end{exercise}
