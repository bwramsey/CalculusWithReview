\begin{exercise}
	Let $r(x) = \left(  \sqrt{\frac{x^3+1}{x+2}} + \frac{4}{x} + 1\right)^{42}$. Compute:

	\[ r'(x) =\answer{  42\left( \sqrt{\frac{x^3+1}{x+2}} + \frac{4}{x} + 1\right)^{41} \left( \frac{1}{2\sqrt{\frac{x^3+1}{x+2}}}\frac{3x^2(x+2)-(x^3+1)(1)}{(x+2)^2} - \frac{4}{x^2}\right)  } \]

\end{exercise}
