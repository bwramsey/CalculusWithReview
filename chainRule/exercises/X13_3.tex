\begin{exercise}
	Let $g(w) = 2(w^2+3)\sqrt{4w-1}$. 
	Compute:

	\[ \dd{w}g(w) =\answer{ 4w\sqrt{4w-1} + \frac{4(w^2+3)}{\sqrt{4w-1}}  } \]
	\begin{hint}
		Which is the `outside operation'?
			\begin{multipleChoice}
				\choice{Sum/Difference}
				\choice[correct]{Product}
				\choice{Quotient}
				\choice{Composition}
			\end{multipleChoice}
	\end{hint}
\end{exercise}
