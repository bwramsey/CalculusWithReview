\begin{exercise}
	Let $g(w) = 2(w^2+3)\sqrt{4w-1}$. 
	Compute:

	\[ \dd{w}g(w) =\answer{ 4w\sqrt{4w-1} + \frac{4(w^2+3)}{\sqrt{4w-1}}  } \]

\end{exercise}
