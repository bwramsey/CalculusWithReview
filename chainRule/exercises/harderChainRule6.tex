\outcome{Recognize a composition of functions.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\begin{document}
\begin{exercise}
Let $P(t) = \sqrt[3]{2} \sqrt{\frac{4}{t ^7}+4 t }$. Compute:

\[
\dd{t}P(t) = \answer{ \sqrt[3]{2} \frac{\left(4-\frac{28}{t^8}\right)}{2 \sqrt{\frac{4}{t^7}+4 t }} }
\]
\end{exercise}
