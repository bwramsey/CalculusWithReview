\begin{exercise}
Let $P(\theta) = \sqrt[3]{2} \sqrt[3]{\frac{4}{\theta ^7}+4 \theta }$. Compute:

\[
\dd{\theta}P(\theta)
\begin{prompt}=\answer{\frac{\sqrt[3]{2} \left(4-\frac{28}{\theta ^8}\right)}{3 \sqrt[3]{\frac{4}{\theta ^7}+4 \theta }^2}}\end{prompt}
\]
\end{exercise}
