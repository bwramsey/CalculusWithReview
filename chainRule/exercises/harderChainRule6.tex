\outcome{Take derivatives of compositions of functions using the chain rule.}

\begin{exercise}
Let $P(s) =4^{-3}\left(\frac{2}{s^7}+3s \right)^{4}$. Compute:

\[
\dd{s}P(s) =\answer{4^{-2}\left(\frac{2}{s^7}+3s \right)^{3}\left(-\frac{14}{s^8} +3\right)}
\]
\end{exercise}
