\begin{exercise}


Evaluate the following limits.
\begin{align*}
	\lim_{x\to\infty} \frac{\sqrt{x}}{\ln(x)} &= \answer{\infty} \\ \\ \\ \\
	\lim_{y\to \infty} (x - \sqrt{x^2+1}) &= \answer{0} \\ \\ \\ \\	
	\lim_{t\to0^+} (1+t)^{\frac{1}{t}} &= \answer{e}
\end{align*}

\end{exercise}
