\begin{exercise}


Find the differential of the following functions.
\begin{align*}
	y &= \sqrt{x+1}\\ \\
	\d y &= \answer{ \frac{1}{2 \sqrt{x+1}} } \d \answer{x} \\ \\ \\ \\
	z &= e^y\cos(2y)\\ \\
	\d z &= \left( \answer{ e^y\cos(2y) - 2 e^y \sin(2y) }\right) \d \answer{y} \\ \\ \\ \\	
	s &= t^4+5t\sin(t) \\ \\
	\d s &= \left(\answer{ 4t^3 + 5\sin(t)+5t\cos(t) } \right) \d \answer{t}
\end{align*}

\end{exercise}
