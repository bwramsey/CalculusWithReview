\begin{exercise}

    Select all correct statements below (For $x > 0$).
    
    \begin{selectAll}
        \choice{$\sqrt{4x+9x^5} = x\sqrt{4+9x^4}$}
        \choice[correct]{$\sqrt{4x+9x^5} = x\sqrt{\frac{4}{x}+9x^3}$}
        \choice{$\sqrt{4x+9x^5} = x\sqrt{\frac{4}{x}+9x^4}$}
        \choice{$\sqrt{4x+9x^5} = x^2\sqrt{\frac{4}{x}+9x^2}$}
        \choice{$\sqrt{4x+9x^5} = x^2\sqrt{\frac{4}{x^2}+9x^2}$}
        \choice[correct]{$\sqrt{4x+9x^5} = x^2\sqrt{\frac{4}{x^3}+9x}$}
        \choice{$\sqrt{4x+9x^5} = x^2\sqrt{\frac{4}{x^3}+9x^3}$}
    \end{selectAll}

	\begin{exercise}
		Evaluate the limits:
		\[ \lim_{x \to \infty} \frac{\sqrt{4x^2+9x+1}}{2x+3} = \answer{1} \]
		
		\[ \lim_{x \to -\infty} \frac{\sqrt{4x^2+9x+1}}{2x+3} = \answer{-1} \]
	\end{exercise}


\end{exercise}
