\begin{exercise}

	Evaluate the limits:
	\begin{align*}
		\lim_{x\to 0} \frac{4x^2-3}{2xe^x+1} &= \answer{-3}\\
		\\
		\lim_{x \to 2} \frac{\sqrt{2\cdot 3^x+3x+1}}{x+2} &= \answer{5/4}\\
		\\
		\lim_{x\to 3} \ln\left( x^2-3x+1\right) &= \answer{0}\\
		\\
		\lim_{x \to -2} \log_{2}\left( 3x^2+2x\right) &= \answer{3}
	\end{align*}

\end{exercise}
