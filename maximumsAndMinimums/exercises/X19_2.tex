	On which intervals is $f$ concave up?
	\[ \left( \, \answer{-2} \, , \, \answer{0} \, \right) \,\, , \,\, \left( \, \answer{1} \, , \, \answer{\infty} \, \right) \]
	How many inflection points does $f$ have? $\answer{3}$.
	\begin{exercise}
