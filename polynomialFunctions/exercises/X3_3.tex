\begin{exercise}
	A polynomial function $p$ has zeroes at $c = -1$, $0$, and $2$ of multiplicities $2$, $1$, and $3$ respectively.  The leading coefficient of $p$ is $-2$.

	What is the degree of the polynomial $p$?
	\[ \text{degree} = \answer{6} \]
	
	As $x$ grows in the positive direction, what happens to the values of $p$?
	\begin{multipleChoice}
		\choice{They get closer and closer to $-2$.}
		\choice{They tend to $\infty$}
		\choice[correct]{They tend to $-\infty$}.
	\end{multipleChoice}

	As $x$ grows in the negative direction, what happens to the values of $p$?
	\begin{multipleChoice}
		\choice{They get closer and closer to $-2$.}
		\choice{They tend to $\infty$}
		\choice[correct]{They tend to $-\infty$}.
	\end{multipleChoice}
\end{exercise}
